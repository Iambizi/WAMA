## 3. Convex Schema

```typescript
// convex/schema.ts
import { defineSchema, defineTable } from "convex/server";
import { v } from "convex/values";

export default defineSchema({
  buyers: defineTable({
    // Identity (advisor-entered, not exposed to AI)
    name: v.string(),
    email: v.string(),
    phone: v.optional(v.string()),

    // Acquisition criteria (used in AI matching)
    sectorInterest: v.array(v.string()),
    budgetMin: v.number(),           // in CAD
    budgetMax: v.number(),
    geography: v.array(v.string()),
    financingType: v.union(
      v.literal("cash"),
      v.literal("financed"),
      v.literal("mixed"),
      v.literal("unknown")
    ),
    acquisitionExperience: v.union(
      v.literal("first_time"),
      v.literal("experienced"),
      v.literal("serial")
    ),
    acquisitionTimeline: v.union(
      v.literal("0_6mo"),
      v.literal("6_12mo"),
      v.literal("12_24mo"),
      v.literal("24mo_plus")
    ),

    // Readiness checklist
    proofOfFundsReviewed: v.boolean(),
    ndaSigned: v.boolean(),
    backgroundCheckComplete: v.boolean(),

    // Qualification
    qualificationStatus: v.union(
      v.literal("pending"),
      v.literal("qualified"),
      v.literal("disqualified")
    ),

    // Internal
    notes: v.optional(v.string()),
    createdAt: v.number(),
    updatedAt: v.number(),
  }),

  sellers: defineTable({
    // Identity (advisor-entered, not exposed to AI)
    name: v.string(),
    email: v.string(),
    phone: v.optional(v.string()),

    // Business profile (used in AI matching — no PII)
    businessName: v.string(),
    sector: v.string(),
    geography: v.string(),
    revenueRange: v.union(
      v.literal("under_500k"),
      v.literal("500k_1m"),
      v.literal("1m_3m"),
      v.literal("3m_5m"),
      v.literal("5m_10m"),
      v.literal("over_10m")
    ),
    ebitdaRange: v.union(
      v.literal("under_100k"),
      v.literal("100k_250k"),
      v.literal("250k_500k"),
      v.literal("500k_1m"),
      v.literal("over_1m")
    ),
    employeeCount: v.union(
      v.literal("1_5"),
      v.literal("6_15"),
      v.literal("16_50"),
      v.literal("51_plus")
    ),
    yearsInOperation: v.number(),
    transactionType: v.union(
      v.literal("full_sale"),
      v.literal("majority"),
      v.literal("minority"),
      v.literal("succession")
    ),
    reasonForSale: v.string(),       // Free text — advisor-summarized, not raw seller input

    // Document readiness checklist (no file storage)
    financialStatementsAvailable: v.boolean(),
    taxReturnsAvailable: v.boolean(),
    leaseDocumentsAvailable: v.boolean(),
    corporateDocumentsAvailable: v.boolean(),
    ndaSigned: v.boolean(),

    // Qualification
    qualificationStatus: v.union(
      v.literal("pending"),
      v.literal("qualified"),
      v.literal("disqualified")
    ),
    readinessScore: v.optional(v.number()),  // 0–100, computed from checklist

    // Internal
    notes: v.optional(v.string()),
    createdAt: v.number(),
    updatedAt: v.number(),
  }),

  matches: defineTable({
    sellerId: v.id("sellers"),
    buyerId: v.id("buyers"),

    // AI output
    aiScore: v.number(),             // 0–100
    aiReasoning: v.string(),         // Plain-language explanation
    aiMatchedCriteria: v.array(v.string()),  // ["sector", "budget", "geography"]

    // Pipeline
    status: v.union(
      v.literal("suggested"),        // AI generated, not yet reviewed
      v.literal("reviewed"),         // Advisor looked at it
      v.literal("approved"),         // Advisor approved introduction
      v.literal("introduced"),       // Parties have been introduced
      v.literal("nda_signed"),
      v.literal("in_discussions"),
      v.literal("closed_won"),
      v.literal("closed_lost"),
      v.literal("rejected")          // Advisor rejected the match
    ),

    advisorNotes: v.optional(v.string()),
    createdAt: v.number(),
    updatedAt: v.number(),
  }).index("by_seller", ["sellerId"])
    .index("by_buyer", ["buyerId"]),

  activityLogs: defineTable({
    action: v.string(),              // "buyer_created", "status_changed", "match_generated", etc.
    entityType: v.union(
      v.literal("buyer"),
      v.literal("seller"),
      v.literal("match")
    ),
    entityId: v.string(),
    entityLabel: v.string(),         // Human-readable: "Buyer #14" or buyer name
    details: v.optional(v.string()), // e.g. "status: pending → qualified"
    createdAt: v.number(),
  }).index("by_created_at", ["createdAt"]),
});
```

```typescript
// convex/buyers.ts — example of auth guard on every mutation
import { mutation, query } from "./_generated/server";
import { v } from "convex/values";

// Helper — throws if not authenticated
async function requireAdvisor(ctx: any) {
  const identity = await ctx.auth.getUserIdentity();
  if (!identity) throw new Error("Unauthorized");
  return identity;
}

export const list = query({
  handler: async (ctx) => {
    await requireAdvisor(ctx);
    return await ctx.db.query("buyers").order("desc").collect();
  },
});

export const create = mutation({
  args: { /* buyer fields */ },
  handler: async (ctx, args) => {
    await requireAdvisor(ctx);
    const now = Date.now();
    const id = await ctx.db.insert("buyers", {
      ...args,
      qualificationStatus: "pending",
      createdAt: now,
      updatedAt: now,
    });
    await ctx.db.insert("activityLogs", {
      action: "buyer_created",
      entityType: "buyer",
      entityId: id,
      entityLabel: args.name,
      createdAt: now,
    });
    return id;
  },
});
```

## 6. AI Matching — Server Route Spec

### Route: `POST /api/match`

**Request body:**
```typescript
{ sellerId: string }
```

**Logic:**
1. Fetch seller by ID from Convex (server SDK)
2. Fetch all buyers where `qualificationStatus === "qualified"`
3. Build de-identified prompt (see below)
4. Call Claude API
5. Parse JSON response
6. Upsert matches into `matches` table
7. Log `match_generated` in `activityLogs`
8. Return match results

### De-identification Rules

**Include in prompt:**
- Internal reference ID (Convex `_id` short form)
- Sector, geography, revenue/EBITDA/budget ranges
- Transaction type, financing type, timeline
- Acquisition experience level
- Readiness checklist booleans

**Never include in prompt:**
- Names
- Emails / phone numbers
- Company names
- Exact financial figures (use ranges only)
- Free-text notes fields

### Prompt Template

```typescript
const systemPrompt = `
You are an M&A match recommendation engine for a boutique SME advisor in Quebec, Canada.
Your job is to analyze a seller profile and a list of qualified buyers, then recommend the best matches.

Rules:
- Only use the structured criteria provided. Do not infer or hallucinate details.
- Score each buyer from 0 to 100 based on fit with the seller.
- Explain your reasoning in plain language (2–3 sentences max per match).
- List which specific criteria aligned (e.g. "sector", "budget", "geography", "timeline").
- Return ONLY valid JSON. No preamble, no markdown.

Response format:
{
  "matches": [
    {
      "buyerRef": "string",
      "score": number,
      "reasoning": "string",
      "matchedCriteria": ["sector", "budget", ...]
    }
  ]
}
`;

const userPrompt = `
SELLER PROFILE:
- Ref: ${sellerRef}
- Sector: ${seller.sector}
- Geography: ${seller.geography}
- Revenue range: ${seller.revenueRange}
- EBITDA range: ${seller.ebitdaRange}
- Transaction type: ${seller.transactionType}
- Employee count: ${seller.employeeCount}
- Years in operation: ${seller.yearsInOperation}
- Reason for sale: ${seller.reasonForSale}

QUALIFIED BUYERS:
${buyers.map(b => `
- Ref: ${b.ref}
  Sectors of interest: ${b.sectorInterest.join(", ")}
  Budget: ${b.budgetMin}–${b.budgetMax} CAD
  Geography: ${b.geography.join(", ")}
  Financing: ${b.financingType}
```

### Claude API Call

```typescript

const client = new Anthropic(); // reads ANTHROPIC_API_KEY from env
```

## 7. Security Model

### Access Rules (enforced in Convex, not just UI)

| Actor | Can read | Can write |
|---|---|---|
| Unauthenticated | Nothing | Nothing |
| Advisor (authenticated) | All tables | All tables |
| System (AI route) | De-identified criteria fields only | matches, activityLogs |

Every Convex query and mutation must call `requireAdvisor(ctx)` as the first line.

### Data Isolation Principles

- Buyers table has no reference to sellers table (no accidental join exposure)
- The AI prompt builder is a separate utility function with an explicit allowlist of fields — not a generic serializer
- `notes` fields on buyers/sellers are never included in AI prompts
- `activityLogs` entries never include raw personal data

## 9. Readiness Score Computation

Computed on the server side when a seller profile is saved or updated. Not stored as a user-editable field.

```typescript
function computeReadinessScore(seller: SellerFields): number {
  const checks = [
    seller.financialStatementsAvailable,
    seller.taxReturnsAvailable,
    seller.leaseDocumentsAvailable,
    seller.corporateDocumentsAvailable,
    seller.ndaSigned,
  ];
  const completed = checks.filter(Boolean).length;
  return Math.round((completed / checks.length) * 100);
}
```

## 11. UI Component Notes

- Use `shadcn/ui` for all form inputs, buttons, badges, cards, dialogs
- Status badges should use color-coded variants: `pending` = gray, `qualified` = green, `disqualified` = red
- Pipeline stages in `/matches/[id]` rendered as a horizontal stepper or segmented control
- Dashboard stats grid: 4 cards — Total Buyers, Total Sellers, Pending Reviews, Active Matches
- Activity feed: chronological list, most recent first, from `activityLogs`

## 12. Build Order (Recommended)

Build in this sequence to always have a working state:

1. **Project scaffold** — repo setup, Convex init, Clerk auth, layout with sidebar
2. **Buyer CRUD** — schema, list page, create form, profile page, status change
3. **Seller CRUD** — same as buyer, add readiness checklist + score
4. **Dashboard** — stats grid (counts from Convex), activity feed
5. **AI matching** — `/api/match` route, prompt builder, match display
6. **Match pipeline** — match detail page, stage advancement
7. **Polish** — loading states, empty states, error handling, copy file cleanup
